#定义一个人名的列表
name = [‘renheng’ , ‘wanghuihui’ , ‘chenyuchen’]

#打印该列表
print(name)

#在尾部添加一个人名，打印
name.append(‘liangqihao’)
print(name)

#在指定位置添加一个人名
name.insert(2,’leikunting’)

#合并两个列表，打印新列表
list = [‘xiaoming’,’xiaohong’,’xiaolan’]
all=name+list
print(all)

#新列表删除一个人名，并打印
del all[1]
print(all)

#找出一指定人名的位置
print(all.index(‘renheng’))

#两种方法遍历列表
for i in range(len(all)):
print(all[i]);
for index,value in enumerate(name):
print(“第%d个元素值是[%s]” %(index ,value));

#对列表进行升序排列，打印
all.sort()
print(all)

#对列表进行降序排列，打印
all.reverse()
print(all)

#产生一个数值递增列表
list1= range(10)
for index,value in enumerate(list1):
print(“list1的第%d个元素是[%s]” %(index,value));

import java.net.InetAddress;
import java.util.Properties;
import java.net.UnknownHostException;
public class search {

public static void main(String[] args) throws Exception {
    InetAddress addr = InetAddress.getLocalHost();  
    String ip=addr.getHostAddress().toString(); //获取本机ip  
    String hostName=addr.getHostName().toString(); //获取本机计算机名称  
    System.out.println("本机ip:"+ip);
    System.out.println("本机名字:"+hostName);

    Properties props=System.getProperties(); //获得系统属性集    
     String osName = props.getProperty("os.name"); //操作系统名称    
     String osArch = props.getProperty("os.arch"); //操作系统构架    
     String osVersion = props.getProperty("os.version"); //操作系统版本  
     System.out.println("操作系统名称:"+osName);
     System.out.println("操作系统构架:"+osArch);
     System.out.println("操作系统版本:"+osVersion);


     //获取www.baidu.com的地址  
     System.out.println(InetAddress.getByName("www.163.com"));

宗旨

Markdown 的目标是实现「易读易写」。

可读性，无论如何，都是最重要的。一份使用 Markdown 格式撰写的文件应该可以直接以纯文本发布，并且看起来不会像是由许多标签或是格式指令所构成。Markdown 语法受到一些既有 text-to-HTML 格式的影响，包括 Setext、atx、Textile、reStructuredText、Grutatext 和 EtText，而最大灵感来源其实是纯文本电子邮件的格式。

总之， Markdown 的语法全由一些符号所组成，这些符号经过精挑细选，其作用一目了然。比如：在文字两旁加上星号，看起来就像强调。Markdown 的列表看起来，嗯，就是列表。Markdown 的区块引用看起来就真的像是引用一段文字，就像你曾在电子邮件中见过的那样。
